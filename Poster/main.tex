    \RTGBlockThin{Motivation}{
        For a smooth projective complex curve $X$, the non-abelian Hodge correspondence can be understood as relating the geometry of the generic fiber and the special fiber inside the moduli space of $\lambda$-connections on $X$
        \begin{center}
            \begin{tikzcd}[ampersand replacement=\&, column sep=large, row sep=huge]
                {\mc{M}_{n, Dol}^0(X)} \& {\mc{M}_{n,Hdg}^0(X)} \& {\mc{M}_{n, DR}^0(X)} \& {\mc{M}_{n,B}^0(X) = Loc_n(X)} \\
                {\{ 0 \}} \& {\Ab^1} \& {\{ 1 \}} \& {} \\
                \arrow[hook, from=1-1, to=1-2]
                \arrow[from=1-1, to=2-1]
                \arrow["\lrcorner"{anchor=center, pos=0.125}, draw=none, from=1-1, to=2-2]
                \arrow[from=1-2, to=2-2]
                \arrow[hook', from=1-3, to=1-2]
                \arrow["\cong", from=1-3, to=1-4]
                \arrow["\lrcorner"{anchor=center, pos=0.125, rotate=-90}, draw=none, from=1-3, to=2-2]
                \arrow[from=1-3, to=2-3]
                \arrow[from=2-1, to=2-2]
                \arrow[from=2-3, to=2-2]
        \end{tikzcd}
        \end{center}
        \vspace{-4.5cm}

        \begin{theorem3*}
        Degree $d$ vector bundles don't admit flat connection. How to get a good notion of $\mc{M}_{n,DR}^d(X)$ that fits into a similar description?
        \end{theorem3*}
    }

\begin{columns}
    \column{.5}
    \RTGBlockThin{Degree $d$ on a curve as degree $0$ on $\mc{X}$}{
        Vector bundles on $\mc{X}$ have a third numerical invariant beyond rank and degree, namely the representation obtained as stalk at the stacky point. We denote by $\lambda(-d)$ the weight $-d$ character of $\mu_n$ and by $n \cdot \lambda(-d)$ the corresponding $n$-dimensional representation.

        \begin{theorem1*}[]
            The twisted pullback and pushforward along the moduli map $\mc{X} \to X$ define equivalences of algebraic stacks
            \[
            \mc{M}_{n, Dol}^d(X) \cong \mc{M}_{n \cdot \lambda(-d), Dol}^0(\mc{X}).
            \]
        \end{theorem1*}

        \begin{theorem2*}
            Deegree $d$ Higgs bundles on $X$ can be understood as degree $0$ Higgs bundles on $\mc{X}$.
        \end{theorem2*}

    }

    \column{.5}
    \RTGBlockThin{Twisted character variety as Betti moduli space on $\mc{X}$}{
        Computing the fundamental group of $\mc{X}$, we see that the Betti moduli space on $\mc{X}$ is
        \[
        \mc{M}_{n, B}(\mc{X}) = \left[ \bigg\{ ((A_i, B_i)_{i = 1}^g, C) \in GL_n^{2g +1} \mymid \prod_{i=1}^{g}[A_i, B_i] = C, \, C^n = Id \bigg\} / GL_n \right].
        \]

        It decomposes with respect to the conjugacy class of $C$, and the component corresponding to $\mc{M}_{n \cdot \lambda(-d), Dol}^0(\mc{X})$ is
        \[
        \mc{M}_{n \cdot \lambda(-d), B}(\mc{X}) = \left[ \bigg\{ (A_i, B_i)_{i = 1}^g \in GL_n^{2g} \mymid \prod_{i=1}^{g}[A_i, B_i] = \mathrm{e}^{-\frac{2 \pi \mathrm{i}}{n} d} Id \bigg\} / GL_n \right].
        \]
        This is exactly the twisted character variety of $X$.
        
        \begin{theorem2*}
        The twisted character variety of $X$ is a connected component of the character variety of $\mc{X}$.
        \end{theorem2*}
    }

\end{columns}

\WhiteBlockSimple{Consequence}{
        We can identify $\mc{M}_{n,DR}^d(X)$ with $\mc{M}_{n \cdot \lambda(-d), DR}^0(\mc{X})$.
        Therefore, the inclusions $\mc{M}_{n, Dol}(\mc{X}) \hookrightarrow \mc{M}_{n, Hdg}(\mc{X})$ and $\mc{M}_{n, DR}(\mc{X}) \hookrightarrow \mc{M}_{n, Hdg}(\mc{X})$ and the (analytic) Riemann-Hilbert correspondence $\mc{M}_{n, DR}(\mc{X}) \cong \mc{M}_{n, B}(\mc{X})$ induce isomorphisms in cohomology
        \[
        H^*(\mc{M}_{n, Dol}^d(X)) \cong H^*(\mc{M}_{n, DR}^d(X)) \cong H^*(\mc{M}_{n, B}^d(X)).
        \]

        \begin{theorem2*}
        The non-abelian Hodge correspondence for coprime $n,d$ on $X$ can be understood from the non-abelian Hodge correspondence for $0$-degree bundles on $\mc{X}$.
        \end{theorem2*}
    }

\begin{columns}
    \column{.5}
    \RTGBlockThin{What about the Hodge structure?}{
        The equivalence between the De Rham and the Betti moduli space is analytic, so it doesn't preserve the Hodge structure.

        \begin{theorem1*}[]
            The tautological class $e_{i,j}$ (i.e. the $j$-th K\"unneth component of $c_i(\mc{E}_{univ})$) on $\mc{M}_{n, Dol}(\mc{X})$ has pure weight $j$. Under the isomorphism $\mc{M}_{n, DR}(\mc{X}) \cong \mc{M}_{n, B}(\mc{X})$ it is mapped to a cohomology class of weight $2i$.
        \end{theorem1*}
    }

    \column{.5}
    \WhiteBlockSimple{Computing the Hodge weights}{    
    To compute the Hodge weights we replicate Shende's argument presented in \cite{Shende}. We construct a simplicial set $\Delta_{\mc{X}}$ realizing a triangulation of $\mc{X}$. We make it into a simplicial scheme and use the equivalence
    \[
    \mc{M}_{n, B}(\mc{X})=\left[ \text{Hom}(\pi_1(\mc{X}, v_0), GL_n) / GL_n \right] \cong \shHom_{SSch}(\Delta_X, BGL_n)
    \]
    and the universal object on $\shHom_{SSch}(\Delta_X, BGL_n)$ to carry out the computation of the weights.
    }

\end{columns}
